# 62048

### 내가 쓴 코드

~~~kotlin
class Solution {
    fun solution(w: Int, h: Int): Long {
    val wLog = w.toLong()
    val hLog = h.toLong()
    var num = 1
    val sLog : Long = if(wLog>hLog){ hLog }else{ wLog }
    for(i in 1..sLog.toInt()){
        if(wLog.toInt()%i == 0&& hLog.toInt()%i == 0){
            num = i
        }
    }
    var answer: Long = wLog*hLog-(wLog+hLog)+num

    return answer
	}
}
~~~
